public class Solution {
    public int reverse(int x) {
        int ret=0;
		int cur=0;
		boolean neg=false;
		
		if(x<0){
			neg=true;
			x=-1*x;
		}
		
		while(x!=0){
			cur=x%10;
			if(ret!=0){
				if((Integer.MAX_VALUE-cur)/ret<10)
					return 0;
				if(neg){
					if(-10<(Integer.MIN_VALUE+cur)/ret)
					return 0;
				}
			}
			ret = ret*10+cur;
			x=x/10;
		}
		
		if(neg){
			ret=-1*ret;
		}
		return ret;
    }
}
